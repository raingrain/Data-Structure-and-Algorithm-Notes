# 按公因数计算最大组件大小

## [952. 按公因数计算最大组件大小](https://leetcode.cn/problems/largest-component-size-by-common-factor/)

> - ***Question***
>   - 给定一个由不同正整数的组成的非空数组 `nums` ，考虑下面的图：
>     - 有 `nums.length` 个节点，按从 `nums[0]` 到 `nums[nums.length - 1]` 标记。
>     - 只有当 `nums[i]` 和 `nums[j]` 共用一个大于 `1` 的公因数时， `nums[i]` 和 `nums[j]` 之间才有一条边。
>   - 返回图中最大连通组件的大小。
>   - ***tips:***
>     - `1 <= nums.length <= 2 * 10^4`
>     - `1 <= nums[i] <= 10^5`
>     - `nums` 中所有值都不同

---

## *Java*

> - ***并查集 + gcd***
>   - 超时。

```java
class Solution {
    
    public int largestComponentSize(int[] nums) {
        UnionFind unionFind = new UnionFind(nums.length);
        // 对于每一个数字
        for (int i = 0; i < nums.length; i++) {
            // 看它后面的数字
            for (int j = i + 1; j < nums.length; j++) {
                // 和它的最大公约数是不是超过1，是就合并
                if (gcd(nums[i], nums[j]) != 1) {
                    unionFind.union(i, j);
                }
            }
        }
        return unionFind.maxSetSize();
    }
    
    // 背诵
    private int gcd(int a, int b) {
        return b == 0 ? a : gcd(b, a % b);
    }
    
    private static class UnionFind {
        
        private final int[] parents;
        private final int[] sizes;
        private final int[] stack;
        
        public UnionFind(int len) {
            parents = new int[len];
            sizes = new int[len];
            stack = new int[len];
            for (int i = 0; i < len; i++) {
                parents[i] = i;
                sizes[i] = 1;
            }
        }
        
        public int maxSetSize() {
            int ans = 0;
            for (int size : sizes) {
                ans = Math.max(ans, size);
            }
            return ans;
        }
        
        private int find(int i) {
            int stackIndex = 0;
            while (i != parents[i]) {
                stack[stackIndex++] = i;
                i = parents[i];
            }
            stackIndex--;
            for (; stackIndex >= 0; stackIndex--) {
                parents[stack[stackIndex]] = i;
            }
            return i;
        }
        
        public void union(int i, int j) {
            int father1 = find(i);
            int father2 = find(j);
            if (father1 != father2) {
                int bigSet = sizes[father1] >= sizes[father2] ? father1 : father2;
                int smallSet = bigSet == father1 ? father2 : father1;
                parents[smallSet] = bigSet;
                sizes[bigSet] = sizes[father1] + sizes[father2];
            }
        }
        
    }
    
}
```

> - ***并查集 + 哈希表***

```java
class Solution {
    
    public int largestComponentSize(int[] nums) {
        UnionFind unionFind = new UnionFind(nums.length);
        // key为一个因子
        // value为它最早在数组中哪个位置的元素上出现过，记录位置
        HashMap<Integer, Integer> factorsMap = new HashMap<>();
        for (int index = 0; index < nums.length; index++) {
            int num = nums[index];
            // 计算根号
            int limit = (int) Math.sqrt(num);
            // 一个数为n，那么我们只需要算出它1~根号n的因子，对应的另外一半可以直接得到
            for (int factor = 1; factor <= limit; factor++) {
                // 遍历1到根号的的数字
                if (num % factor == 0) {
                    // 是因子
                    if (factor != 1) {
                        // 不是1
                        if (!factorsMap.containsKey(factor)) {
                            // 表中没有就加记录
                            factorsMap.put(factor, index);
                        } else {
                            // 有的话就和最早出现的那个位置的元素所在集合合并
                            unionFind.union(factorsMap.get(factor), index);
                        }
                    }
                    // 对应的另外一个因子，重复上述操作
                    int anotherFactor = num / factor;
                    if (anotherFactor != 1) {
                        if (!factorsMap.containsKey(anotherFactor)) {
                            factorsMap.put(anotherFactor, index);
                        } else {
                            unionFind.union(factorsMap.get(anotherFactor), index);
                        }
                    }
                }
            }
        }
        return unionFind.maxSetSize();
    }
    
    private static class UnionFind {
        
        // 父元素数组
        private final int[] parents;
        // 集合大小数组
        private final int[] sizes;
        // 栈
        private final int[] stack;
        
        public UnionFind(int len) {
            parents = new int[len];
            sizes = new int[len];
            stack = new int[len];
            // 初始时每个位置都是一个集合
            for (int i = 0; i < len; i++) {
                parents[i] = i;
                sizes[i] = 1;
            }
        }
        
        // 返回最大连通组件大小
        // 一个连通组件就是一个集合
        // 查谁的集合最大
        public int maxSetSize() {
            int ans = 0;
            for (int size : sizes) {
                ans = Math.max(ans, size);
            }
            return ans;
        }
        
        private int find(int i) {
            int stackIndex = 0;
            while (i != parents[i]) {
                stack[stackIndex++] = i;
                i = parents[i];
            }
            stackIndex--;
            for (; stackIndex >= 0; stackIndex--) {
                parents[stack[stackIndex]] = i;
            }
            return i;
        }
        
        public void union(int i, int j) {
            int father1 = find(i);
            int father2 = find(j);
            if (father1 != father2) {
                int bigSet = sizes[father1] >= sizes[father2] ? father1 : father2;
                int smallSet = bigSet == father1 ? father2 : father1;
                parents[smallSet] = bigSet;
                sizes[bigSet] = sizes[father1] + sizes[father2];
            }
        }
        
    }
    
}
```

> - ***并查集 + 质因子分解***

```java
class Solution {

    // 展示分解质因子过程
    public static void main(String[] args) {
        int n = 4012100;
        f(n);
    }

    // 打印所有n的质因子，时间复杂度O(根号n)
    public static void f(int n) {
        for (int i = 2; i * i <= n; i++) {
            if (n % i == 0) {
                System.out.println(i);
                while (n % i == 0) {
                    n /= i;
                }
            }
        }
        if (n > 1) {
            System.out.println(n);
        }
    }

    // 按公因数计算最大组件大小
    // 给定一个由不同正整数的组成的非空数组 nums
    // 如果 nums[i] 和 nums[j] 有一个大于1的公因子，那么这两个数之间有一条无向边
    // 返回 nums中最大连通组件的大小。
    // 测试链接 : https://leetcode.cn/problems/largest-component-size-by-common-factor/
    // 提交以下代码，可以通过所有测试用例

    public static int MAXV = 100001;

    // factors[a] = b
    // a这个质数因子，最早被下标b的数字拥有
    public static int[] factors = new int[MAXV];

    // 并查集模版
    public static int MAXN = 20001;

    public static int[] father = new int[MAXN];

    public static int[] size = new int[MAXN];

    public static int n;

    public static void build() {
        for (int i = 0; i < n; i++) {
            father[i] = i;
            size[i] = 1;
        }
        Arrays.fill(factors, -1);
    }

    public static int find(int i) {
        if (i != father[i]) {
            father[i] = find(father[i]);
        }
        return father[i];
    }

    public static void union(int x, int y) {
        int fx = find(x);
        int fy = find(y);
        if (fx != fy) {
            father[fx] = fy;
            size[fy] += size[fx];
        }
    }

    public static int maxSize() {
        int ans = 0;
        for (int i = 0; i < n; i++) {
            ans = Math.max(ans, size[i]);
        }
        return ans;
    }

    // 正式方法
    // 时间复杂度O(n * 根号v)
    public static int largestComponentSize(int[] arr) {
        n = arr.length;
        build();
        for (int i = 0, x; i < n; i++) {
            x = arr[i];
            for (int j = 2; j * j <= x; j++) {
                if (x % j == 0) {
                    if (factors[j] == -1) {
                        factors[j] = i;
                    } else {
                        union(factors[j], i);
                    }
                    while (x % j == 0) {
                        x /= j;
                    }
                }
            }
            if (x > 1) {
                if (factors[x] == -1) {
                    factors[x] = i;
                } else {
                    union(factors[x], i);
                }
            }
        }
        return maxSize();
    }

}
```
