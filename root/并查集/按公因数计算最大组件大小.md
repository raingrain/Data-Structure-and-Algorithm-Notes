# 按公因数计算最大组件大小

## [952. 按公因数计算最大组件大小](https://leetcode.cn/problems/largest-component-size-by-common-factor/)

> - ***Question***
>   - 给定一个由不同正整数的组成的非空数组 `nums` ，考虑下面的图：
>     - 有 `nums.length` 个节点，按从 `nums[0]` 到 `nums[nums.length - 1]` 标记。
>     - 只有当 `nums[i]` 和 `nums[j]` 共用一个大于 `1` 的公因数时， `nums[i]` 和 `nums[j]` 之间才有一条边。
>   - 返回图中最大连通组件的大小。
>   - ***tips:***
>     - `1 <= nums.length <= 2 * 10^4`
>     - `1 <= nums[i] <= 10^5`
>     - `nums` 中所有值都不同

---

## *Java*

> - ***并查集 + gcd***
>   - 超时。

```java
class Solution {
    
    public int largestComponentSize(int[] nums) {
        UnionFind unionFind = new UnionFind(nums.length);
        // 对于每一个数字
        for (int i = 0; i < nums.length; i++) {
            // 看它后面的数字
            for (int j = i + 1; j < nums.length; j++) {
                // 和它的最大公约数是不是超过1，是就合并
                if (gcd(nums[i], nums[j]) != 1) {
                    unionFind.union(i, j);
                }
            }
        }
        return unionFind.maxSetSize();
    }
    
    // 背诵
    private int gcd(int a, int b) {
        return b == 0 ? a : gcd(b, a % b);
    }
    
    private static class UnionFind {
        
        private final int[] parents;
        private final int[] sizes;
        private final int[] stack;
        
        public UnionFind(int len) {
            parents = new int[len];
            sizes = new int[len];
            stack = new int[len];
            for (int i = 0; i < len; i++) {
                parents[i] = i;
                sizes[i] = 1;
            }
        }
        
        public int maxSetSize() {
            int ans = 0;
            for (int size : sizes) {
                ans = Math.max(ans, size);
            }
            return ans;
        }
        
        private int find(int i) {
            int stackIndex = 0;
            while (i != parents[i]) {
                stack[stackIndex++] = i;
                i = parents[i];
            }
            stackIndex--;
            for (; stackIndex >= 0; stackIndex--) {
                parents[stack[stackIndex]] = i;
            }
            return i;
        }
        
        public void union(int i, int j) {
            int father1 = find(i);
            int father2 = find(j);
            if (father1 != father2) {
                int bigSet = sizes[father1] >= sizes[father2] ? father1 : father2;
                int smallSet = bigSet == father1 ? father2 : father1;
                parents[smallSet] = bigSet;
                sizes[bigSet] = sizes[father1] + sizes[father2];
            }
        }
        
    }
    
}
```

> - ***并查集 + 哈希表***

```java
class Solution {
    
    public int largestComponentSize(int[] nums) {
        UnionFind unionFind = new UnionFind(nums.length);
        // key为一个因子
        // value为它最早在数组中哪个位置的元素上出现过，记录位置
        HashMap<Integer, Integer> factorsMap = new HashMap<>();
        for (int index = 0; index < nums.length; index++) {
            int num = nums[index];
            // 计算根号
            int limit = (int) Math.sqrt(num);
            // 一个数为n，那么我们只需要算出它1~根号n的因子，对应的另外一半可以直接得到
            for (int factor = 1; factor <= limit; factor++) {
                // 遍历1到根号的的数字
                if (num % factor == 0) {
                    // 是因子
                    if (factor != 1) {
                        // 不是1
                        if (!factorsMap.containsKey(factor)) {
                            // 表中没有就加记录
                            factorsMap.put(factor, index);
                        } else {
                            // 有的话就和最早出现的那个位置的元素所在集合合并
                            unionFind.union(factorsMap.get(factor), index);
                        }
                    }
                    // 对应的另外一个因子，重复上述操作
                    int anotherFactor = num / factor;
                    if (anotherFactor != 1) {
                        if (!factorsMap.containsKey(anotherFactor)) {
                            factorsMap.put(anotherFactor, index);
                        } else {
                            unionFind.union(factorsMap.get(anotherFactor), index);
                        }
                    }
                }
            }
        }
        return unionFind.maxSetSize();
    }
    
    private static class UnionFind {
        
        // 父元素数组
        private final int[] parents;
        // 集合大小数组
        private final int[] sizes;
        // 栈
        private final int[] stack;
        
        public UnionFind(int len) {
            parents = new int[len];
            sizes = new int[len];
            stack = new int[len];
            // 初始时每个位置都是一个集合
            for (int i = 0; i < len; i++) {
                parents[i] = i;
                sizes[i] = 1;
            }
        }
        
        // 返回最大连通组件大小
        // 一个连通组件就是一个集合
        // 查谁的集合最大
        public int maxSetSize() {
            int ans = 0;
            for (int size : sizes) {
                ans = Math.max(ans, size);
            }
            return ans;
        }
        
        private int find(int i) {
            int stackIndex = 0;
            while (i != parents[i]) {
                stack[stackIndex++] = i;
                i = parents[i];
            }
            stackIndex--;
            for (; stackIndex >= 0; stackIndex--) {
                parents[stack[stackIndex]] = i;
            }
            return i;
        }
        
        public void union(int i, int j) {
            int father1 = find(i);
            int father2 = find(j);
            if (father1 != father2) {
                int bigSet = sizes[father1] >= sizes[father2] ? father1 : father2;
                int smallSet = bigSet == father1 ? father2 : father1;
                parents[smallSet] = bigSet;
                sizes[bigSet] = sizes[father1] + sizes[father2];
            }
        }
        
    }
    
}
```
