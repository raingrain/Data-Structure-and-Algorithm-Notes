# 打砖块

## [803. 打砖块](https://leetcode.cn/problems/bricks-falling-when-hit/)

> - ***Question***
>   - 有一个 `m x n` 的二元网格 `grid` ，其中 `1` 表示砖块， `0` 表示空白。砖块稳定（不会掉落）的前提是：
>     - 一块砖直接连接到网格的顶部。
>     - 或者至少有一块相邻即 `4` 个方向之一砖块稳定不会掉落时。
>   - 给你一个数组 `hits` ，这是需要依次消除砖块的位置。每当消除 `hits[i] = (rowi, coli)` 位置上的砖块时，对应位置的砖块（若存在）会消失，然后其他的砖块可能因为这一消除操作而掉落。一旦砖块掉落，它会 立即从网格 `grid` 中消失（即它不会落在其他稳定的砖块上）。
>   - 返回一个数组 `result` ，其中 `result[i]` 表示第 `i` 次消除操作对应掉落（不包括击中的那个）的砖块数目。
>   - 注意，消除可能指向是没有砖块的空白位置，如果发生这种情况，则没有砖块掉落。
>   - ***tips:***
>     - `m == grid.length`
>     - `n == grid[i].length`
>     - `1 <= m, n <= 200`
>     - `grid[i][j]` 为 `0` 或 `1`
>     - `1 <= hits.length <= 4 * 10^4`
>     - `hits[i].length == 2`
>     - `0 <= xi <= m - 1`
>     - `0 <= yi <= n - 1`
>     - 所有 `(xi, yi)` 互不相同

---

## *Java*

> - ***并查集***

```java
class Solution {
    
    // 1      2
    // 1      1
    // 2 1 1  1 2 1
    // 击中的顺序是(3 0) (3 5) (0 4)
    // 答案是2 1 3
    // 我们逆序思考
    // 我们先把要击中的位置值改成2，然后创建并查集
    // 我们逆序遍历击中顺序，现在有2个砖块和天花板连接
    // 此时将(0 4)变成1，有5个砖块和天花板相连，得到结果为2
    // 然后将(3 5)变成1，有7个砖块和天花板相连，得到结果为1
    // 然后将(3 0)变成1，有10个砖块和天花板相连，得到结果为2
    
    public int[] hitBricks(int[][] grid, int[][] hits) {
        for (int[] hit : hits) {
            // 要击中的位置上如果有砖块就变成1
            if (grid[hit[0]][hit[1]] == 1) {
                grid[hit[0]][hit[1]] = 2;
            }
        }
        // 创建并查集
        UnionFind unionFind = new UnionFind(grid);
        int[] ans = new int[hits.length];
        // 逆序遍历求答案
        for (int i = hits.length - 1; i >= 0; i--) {
            // 必须击中砖块才管，不击中不管它
            if (grid[hits[i][0]][hits[i][1]] == 2) {
                ans[i] = unionFind.finger(hits[i][0], hits[i][1]);
            }
        }
        return ans;
    }
    
    private static class UnionFind {
        
        // 区域有N行
        private int N;
        // 区域有M列
        private int M;
        /// 原始区域
        private int[][] grid;
        // 有多少个砖块与天花板相连接
        private int cellingAll;
        // 下面全是一维数组是因为做了拉伸
        // 有哪些集合是和天花板相连接的集合
        private boolean[] cellingSet;
        // 查询一个节点的父节点是谁
        private int[] fatherMap;
        // 一个集合的大小
        private int[] sizeMap;
        // 查询父节点时需要的栈
        private int[] stack;
        
        public UnionFind(int[][] matrix) {
            // 初始化各个变量
            initSpace(matrix);
            // 将所有的相连的1的集合进行合并
            initConnect();
        }
        
        private void initSpace(int[][] matrix) {
            // 矩阵赋值
            grid = matrix;
            N = grid.length;
            M = grid[0].length;
            // 总共有多少个元素
            int all = N * M;
            cellingAll = 0;
            cellingSet = new boolean[all];
            fatherMap = new int[all];
            sizeMap = new int[all];
            stack = new int[all];
            // 为每一个砖块创建并查集
            for (int row = 0; row < N; row++) {
                for (int col = 0; col < M; col++) {
                    if (grid[row][col] == 1) {
                        // 获取编号
                        int index = row * M + col;
                        // 每一个集合的头节点的父节点是自己
                        fatherMap[index] = index;
                        // 初始大小为1
                        sizeMap[index] = 1;
                        // 如果是天花板集合，还要再设置信息
                        if (row == 0) {
                            // 是天花板集合
                            cellingSet[index] = true;
                            // 数量++
                            cellingAll++;
                        }
                    }
                }
            }
        }
        
        // 初始时将所有单独的1集合进行连接
        private void initConnect() {
            // 暴力遍历每一个点，查看上下左右
            for (int row = 0; row < N; row++) {
                for (int col = 0; col < M; col++) {
                    union(row, col, row - 1, col);
                    union(row, col, row + 1, col);
                    union(row, col, row, col - 1);
                    union(row, col, row, col + 1);
                }
            }
        }
        
        // 合并过程
        private void union(int row1, int col1, int row2, int col2) {
            // 两个位置都不越界且值都是1
            if (isValid(row1, col1) && isValid(row2, col2)) {
                // 分别找到所在集合的父节点
                int father1 = find(row1, col1);
                int father2 = find(row2, col2);
                // 如果两个位置不在一个集合中
                if (father1 != father2) {
                    // 获取集合大小，小集合挂大集合
                    int size1 = sizeMap[father1];
                    int size2 = sizeMap[father2];
                    // 获取这个集合是不是天花板集合
                    boolean status1 = cellingSet[father1];
                    boolean status2 = cellingSet[father2];
                    if (size1 <= size2) {
                        // 挂载
                        fatherMap[father1] = father2;
                        // 被挂载的集合大小更新
                        sizeMap[father2] = size1 + size2;
                        // 只要有一个是天花板集合另外一个不是就改
                        // 都不是或者都是就不改
                        if (status1 ^ status2) {
                            // 被挂载的变成true
                            cellingSet[father2] = true;
                            // 哪个是天花板集合，原先就被计数了，要加上不是的那个
                            cellingAll += status1 ? size2 : size1;
                        }
                    } else {
                        fatherMap[father2] = father1;
                        sizeMap[father1] = size1 + size2;
                        if (status1 ^ status2) {
                            cellingSet[father1] = true;
                            cellingAll += status1 ? size2 : size1;
                        }
                    }
                    
                }
            }
        }
        
        // 找到一个节点的父节点
        private int find(int row, int col) {
            // 栈顶指针
            int stackSize = 0;
            int index = row * M + col;
            // 集合中头节点的父节点是自己
            // 说明还没有来到头节点
            while (index != fatherMap[index]) {
                stack[stackSize++] = index;
                index = fatherMap[index];
            }
            // 找到了，拉平回填
            while (stackSize != 0) {
                fatherMap[stack[--stackSize]] = index;
            }
            return index;
        }
        
        // 给定炮弹打中的位置，返回掉落多少砖块
        private int finger(int row, int col) {
            // 打中的位置复原
            // 即原来值为3的节点值变成1，创建新的节点
            grid[row][col] = 1;
            int cur = row * M + col;
            if (row == 0) {
                cellingSet[cur] = true;
                cellingAll++;
            }
            fatherMap[cur] = cur;
            sizeMap[cur] = 1;
            // 记一下之前的和天花板相连的节点数
            int pre = cellingAll;
            // 合并后
            union(row, col, row - 1, col);
            union(row, col, row + 1, col);
            union(row, col, row, col - 1);
            union(row, col, row, col + 1);
            int now = cellingAll;
            // 可以举具体的例子讨论
            if (row == 0) {
                return now - pre;
            } else {
                return now == pre ? 0 : now - pre - 1;
            }
        }
        
        // 判断一个点是否越界且值是不是1
        private boolean isValid(int row, int col) {
            return row >= 0 && row < N && col >= 0 && col < M && grid[row][col] == 1;
        }
        
    }
    
}
```
